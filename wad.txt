WebForm1.aspx.cs

using System;
using System.Web.UI;

namespace MCA_P3
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
        }

        protected void btnCalculate_Click(object sender, EventArgs e)
        {
            string enroll = txtEnroll.Text;
            string name = txtName.Text;

            int s1 = Convert.ToInt32(txtSub1.Text);
            int s2 = Convert.ToInt32(txtSub2.Text);
            int s3 = Convert.ToInt32(txtSub3.Text);
            int s4 = Convert.ToInt32(txtSub4.Text);
            int s5 = Convert.ToInt32(txtSub5.Text);

            int total = s1 + s2 + s3 + s4 + s5;
            float percentage = total / 5.0f;

            string grade = "";

            if (percentage >= 90)
                grade = "A+";
            else if (percentage >= 80)
                grade = "B";
            else if (percentage >= 70)
                grade = "C";
            else if (percentage >= 60)
                grade = "D";
            else if (percentage >= 50)
                grade = "E";
            else
                grade = "Fail";

            lblResult.Text = "Enrollment: " + enroll + "<br/>" +
                             "Name: " + name + "<br/>" +
                             "Total: " + total + "<br/>" +
                             "Percentage: " + percentage.ToString("F2") + "%<br/>" +
                             "Grade: " + grade;
        }
    }
}

{
    public partial class WebForm1 : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {

        }
        protected void Res(object sender, EventArgs e)
        {
            StringBuilder s = new StringBuilder();
            s.Append("<b>Name:</b>" + name.Text + "</br>");
            s.Append("<b>Date of Birth:</b>" + dob.Text + "</br>");
            s.Append("<b>Gender:</b>" + gen.SelectedValue + "</br>");
            s.Append("<b>Email:</b>" + email.Text + "</br>");

            s.Append("<b>Address:</b>" + adrs.Text + "</br>");
            s.Append("<b>Pin Code:</b>" + pin.Text + "</br>");

            s.Append("<b>Hobbies:</b>");
            foreach (var item in hobbies.Items)
            {
                var litem = (System.Web.UI.WebControls.ListItem)item;
                if (litem.Selected)
                {
                    s.Append(litem.Text + "");
                }
            }
            lblRes.Text = s.ToString();
            lblRes.Visible = true;
        }
    }

            ValidationSettings.UnobtrusiveValidationMode = UnobtrusiveValidationMode.None;

        {
            if (Page.IsValid)

using System;
using System.Configuration;
using System.Data;
using System.Data.SqlClient;
using System.Linq;
using System.Web.UI;

namespace WebApplication17
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        // Connection string name (set this in web.config)
        string cs = ConfigurationManager.ConnectionStrings["trainConn"].ConnectionString;

        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                ddlMethod.Items.Clear();
                ddlMethod.Items.Add(new System.Web.UI.WebControls.ListItem("Stored Procedure", "SP"));
                ddlMethod.Items.Add(new System.Web.UI.WebControls.ListItem("LINQ", "LINQ"));
            }
        }

        // Helper: parse date safely
        private bool TryParseDate(string s, out DateTime dt)
        {
            return DateTime.TryParse(s, out dt);
        }

        // INSERT
        protected void btnInsert_Click(object sender, EventArgs e)
        {
            lblMessage.Text = "";

            if (ddlMethod.SelectedValue == "SP")
            {
                using (SqlConnection con = new SqlConnection(cs))
                using (SqlCommand cmd = new SqlCommand("sp_InsertTrain", con))
                {
                    cmd.CommandType = CommandType.StoredProcedure;
                    DateTime at, dt;

                    cmd.Parameters.AddWithValue("@train_name", txtName.Text.Trim());
                    cmd.Parameters.AddWithValue("@train_type", txtType.Text.Trim());
                    cmd.Parameters.AddWithValue("@arrival_time", TryParseDate(txtArrival.Text.Trim(), out at) ? (object)at : DBNull.Value);
                    cmd.Parameters.AddWithValue("@departure_time", TryParseDate(txtDeparture.Text.Trim(), out dt) ? (object)dt : DBNull.Value);
                    cmd.Parameters.AddWithValue("@start_location", txtStart.Text.Trim());
                    cmd.Parameters.AddWithValue("@end_location", txtEnd.Text.Trim());

                    con.Open();
                    cmd.ExecuteNonQuery();
                }

                lblMessage.Text = "Inserted using Stored Procedure.";
            }
            else
            {
                var db = new TrainDataDataContext(ConfigurationManager.ConnectionStrings["trainConn"].ConnectionString);
                var t = new train_info
                {
                    train_name = txtName.Text.Trim(),
                    train_type = txtType.Text.Trim(),
                    start_location = txtStart.Text.Trim(),
                    end_location = txtEnd.Text.Trim()
                };

                DateTime at, dt;
                if (TryParseDate(txtArrival.Text.Trim(), out at)) t.arrival_time = at;
                if (TryParseDate(txtDeparture.Text.Trim(), out dt)) t.departure_time = dt;

                db.train_infos.InsertOnSubmit(t);
                db.SubmitChanges();
                lblMessage.Text = "Inserted using LINQ.";
            }

            ClearFields();
            btnView_Click(null, null);
        }

        // UPDATE
        protected void btnUpdate_Click(object sender, EventArgs e)
        {
            lblMessage.Text = "";
            int id;

            if (!int.TryParse(txtId.Text.Trim(), out id))
            {
                lblMessage.Text = "Enter valid Train ID to update.";
                return;
            }

            if (ddlMethod.SelectedValue == "SP")
            {
                using (SqlConnection con = new SqlConnection(cs))
                using (SqlCommand cmd = new SqlCommand("sp_UpdateTrain", con))
                {
                    cmd.CommandType = CommandType.StoredProcedure;
                    DateTime at, dt;

                    cmd.Parameters.AddWithValue("@train_id", id);
                    cmd.Parameters.AddWithValue("@train_name", txtName.Text.Trim());
                    cmd.Parameters.AddWithValue("@train_type", txtType.Text.Trim());
                    cmd.Parameters.AddWithValue("@arrival_time", TryParseDate(txtArrival.Text.Trim(), out at) ? (object)at : DBNull.Value);
                    cmd.Parameters.AddWithValue("@departure_time", TryParseDate(txtDeparture.Text.Trim(), out dt) ? (object)dt : DBNull.Value);
                    cmd.Parameters.AddWithValue("@start_location", txtStart.Text.Trim());
                    cmd.Parameters.AddWithValue("@end_location", txtEnd.Text.Trim());

                    con.Open();
                    cmd.ExecuteNonQuery();
                }

                lblMessage.Text = "Updated using Stored Procedure.";
            }
            else
            {
                var db = new TrainDataDataContext(ConfigurationManager.ConnectionStrings["trainConn"].ConnectionString);
                var train = db.train_infos.FirstOrDefault(t => t.train_id == id);
                if (train == null)
                {
                    lblMessage.Text = "No train found with that ID.";
                    return;
                }

                train.train_name = txtName.Text.Trim();
                train.train_type = txtType.Text.Trim();

                DateTime at, dt;
                if (TryParseDate(txtArrival.Text.Trim(), out at)) train.arrival_time = at;
                if (TryParseDate(txtDeparture.Text.Trim(), out dt)) train.departure_time = dt;

                train.start_location = txtStart.Text.Trim();
                train.end_location = txtEnd.Text.Trim();

                db.SubmitChanges();
                lblMessage.Text = "Updated using LINQ.";
            }

            ClearFields();
            btnView_Click(null, null);
        }

        // DELETE
        protected void btnDelete_Click(object sender, EventArgs e)
        {
            lblMessage.Text = "";
            int id;

            if (!int.TryParse(txtId.Text.Trim(), out id))
            {
                lblMessage.Text = "Enter valid Train ID to delete.";
                return;
            }

            if (ddlMethod.SelectedValue == "SP")
            {
                using (SqlConnection con = new SqlConnection(cs))
                using (SqlCommand cmd = new SqlCommand("sp_DeleteTrain", con))
                {
                    cmd.CommandType = CommandType.StoredProcedure;
                    cmd.Parameters.AddWithValue("@train_id", id);
                    con.Open();
                    cmd.ExecuteNonQuery();
                }

                lblMessage.Text = "Deleted using Stored Procedure.";
            }
            else
            {
                var db = new TrainDataDataContext(ConfigurationManager.ConnectionStrings["trainConn"].ConnectionString);
                var train = db.train_infos.FirstOrDefault(t => t.train_id == id);
                if (train == null)
                {
                    lblMessage.Text = "No train found with that ID.";
                    return;
                }

                db.train_infos.DeleteOnSubmit(train);
                db.SubmitChanges();
                lblMessage.Text = "Deleted using LINQ.";
            }

            ClearFields();
            btnView_Click(null, null);
        }

        // SEARCH

            LoadXmlData();

            // Clear inputs 
            txtId.Text = txtName.Text = txtDept.Text = "";
        }

BookController.cs

using MCA_P19.Models;
using System;
using System.Collections.Generic;
using System.Data.Entity;
using System.Linq;
using System.Web;
using System.Web.Mvc;

namespace MCA_P19.Controllers
{
    public class BookController : Controller
    {
        private BookDBContext db = new BookDBContext();

        // View All
        public ActionResult Index()
        {
            return View(db.Books.ToList());
        }

        // Insert
        public ActionResult Create()
        {
            return View();
        }

        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult Create(Book b)
        {
            if (ModelState.IsValid)
            {
                db.Books.Add(b);
                db.SaveChanges();
                return RedirectToAction("Index");
            }
            return View(b);
        }

        // Search
        public ActionResult Search(string name)
        {
            var result = db.Books
                           .Where(x => x.book_name.Contains(name))
                           .ToList();
            return View("Index", result);
        }

        // Update
        public ActionResult Edit(int id)
        {
            var book = db.Books.Find(id);
            if (book == null) return HttpNotFound();
            return View(book);
        }

        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult Edit(Book b)
        {
            if (ModelState.IsValid)
            {
                db.Entry(b).State = EntityState.Modified;
                db.SaveChanges();
                return RedirectToAction("Index");
            }
            return View(b);
        }

        // Delete GET
        public ActionResult Delete(int id)
        {
            var book = db.Books.Find(id);
            if (book == null) return HttpNotFound();
            return View(book);
        }

        // Delete POST
        [HttpPost, ActionName("Delete")]
        [ValidateAntiForgeryToken]
        public ActionResult DeleteConfirmed(int id)
        {
            var book = db.Books.Find(id);
            db.Books.Remove(book);
            db.SaveChanges();
            return RedirectToAction("Index");
        }
    }
}
